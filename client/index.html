const express = require("express"); const http = require("http"); const cors = require("cors"); const { Server } = require("socket.io"); const jwt = require("jsonwebtoken"); require("dotenv").config(); const db = require("./db"); const { router: authRouter } = require("./auth"); const app = express(); app.use(cors()); app.use(express.json()); const server = http.createServer(app); const io = new Server(server, { cors: { origin: "*" } }); const JWT_SECRET = process.env.JWT_SECRET || "dev_secret_change_me"; /* ---------------- AUTH ROUTES ---------------- */ app.use("/auth", authRouter); app.get("/", (req, res) => { res.send("InstantFix backend running"); }); /* ---------------- SOCKET AUTH ---------------- */ io.use((socket, next) => { try { const token = socket.handshake.auth?.token; if (!token) return next(new Error("No token provided")); const decoded = jwt.verify(token, JWT_SECRET); socket.user = decoded; next(); } catch (err) { next(new Error("Invalid token")); } }); /* ---------------- SOCKET CONNECTION ---------------- */ io.on("connection", (socket) => { console.log("user connected:", socket.user?.email); /* SEND ALL JOBS FROM DB */ db.all("SELECT * FROM jobs", [], (err, rows) => { socket.emit("jobs:init", rows || []); }); /* ---------------- CREATE JOB ---------------- */ socket.on("job:create", (data) => { if (socket.user.role !== "client") return; const job = { id: Date.now().toString(), title: data.title, description: data.description, location: data.location, status: "open", clientId: socket.user.id, helperId: null }; db.run( `INSERT INTO jobs (id, title, description, location, status, clientId, helperId) VALUES (?, ?, ?, ?, ?, ?, ?)`, [ job.id, job.title, job.description, job.location, job.status, job.clientId, job.helperId ] ); io.emit("job:new", job); }); /* ---------------- ACCEPT JOB ---------------- */ socket.on("job:accept", ({ jobId }) => { if (socket.user.role !== "helper") return; db.get("SELECT * FROM jobs WHERE id = ?", [jobId], (err, job) => { if (!job || job.status !== "open") return; db.run( "UPDATE jobs SET status = ?, helperId = ? WHERE id = ?", ["accepted", socket.user.id, jobId] ); io.emit("job:update", { ...job, status: "accepted", helperId: socket.user.id }); }); }); /* ---------------- COMPLETE JOB ---------------- */ socket.on("job:done", ({ jobId }) => { db.get("SELECT * FROM jobs WHERE id = ?", [jobId], (err, job) => { if (!job) return; const isHelper = job.helperId === socket.user.id; const isClient = job.clientId === socket.user.id; if (!isHelper && !isClient) return; db.run( "UPDATE jobs SET status = ? WHERE id = ?", ["done", jobId] ); io.emit("job:update", { ...job, status: "done" }); }); }); /* ---------------- DISCONNECT ---------------- */ socket.on("disconnect", () => { console.log("user disconnected:", socket.user?.email || socket.id); }); }); /* ---------------- START SERVER ---------------- */ const PORT = process.env.PORT || 3000; server.listen(PORT, () => { console.log("InstantFix running on port", PORT); });
